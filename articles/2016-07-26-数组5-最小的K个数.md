【题目】输入n个整数，找出其中最小的K个数。例如输入4,5,1,6,2,7,3,8这8个数字，则最小的4个数字是1,2,3,4,。

方法三：使用堆排序。每次返回一个最小值。时间复杂度O（nlogk）

```
import java.util.*;

public class Solution {
    //使用快排，当返回的中枢轴的值恰好为k时，左边的数均比第k个数值小（不一定是排序的），此时K左边的数即为所求、
    public ArrayList<Integer> GetLeastNumbers_Solution(int [] input, int k) {
        ArrayList<Integer> list = new ArrayList<Integer>();
        if(input == null || k <= 0 || k > input.length ){
            return list;
        }
        
        for(int i = input.length/2 - 1; i >= 0; i--){
            heapAdjust(input, i, input.length - 1);
        }
        for(int j = input.length - 1; j >= input.length - k; j--){
            list.add(input[0]);
            Swap(input, 0, j);
            heapAdjust(input, 0, j - 1);
        }
        
        return list ;
        
    }
    
    //构造大顶堆，升序排序
   private void heapAdjust(int[] arr, int index, int len){
       int temp = arr[index];
       for(int j = 2 * index + 1; j < len; j = j * 2 + 1){
           if(j < len && arr[j] > arr[j + 1] ){
               j++;
           }
           if(arr[j] >= temp){
               break;
           }
           arr[index] = arr[j];
           index = j;
       }
       
       arr[index] = temp;
   }
    
    
    private void Swap(int[] arr, int low, int high) {
		int temp = arr[low];
		arr[low] = arr[high];
		arr[high] = temp;
	}
}
```

  
  

方法二：使用快排。当返回的中枢轴为K时，则中枢轴左边的数都比第K个数小，即为所求的最小的K个数。时间复杂度O（n）

```
import java.util.*;

public class Solution {
    //使用快排，当返回的中枢轴的值恰好为k时，左边的数均比第k个数值小（不一定是排序的），此时K左边的数即为所求、
    public ArrayList<Integer> GetLeastNumbers_Solution(int [] input, int k) {
        ArrayList<Integer> list = new ArrayList<Integer>();
        if(input == null || k <= 0 || k > input.length ){
            return list;
        }
        
        int low = 0;
        int high = input.length - 1;
        int pivot = Partition(input, low, high);
        while(pivot != k - 1){
            if(pivot > k - 1){
                pivot = Partition(input, pivot + 1, high);
            } else{
                pivot = Partition(input, low, pivot - 1);
            }
        }
        
        for(int i = 0; i < k; i++){
            list.add(input[i]);
        }
        
        Collections.sort(list);
        return list ;
        
    }
    
    
    private int Partition(int[] arr, int low, int high){
        int pivotKey = arr[low];
        while(low < high){
            while(low < high && arr[high] >= pivotKey){
                high--;
            }
            Swap(arr, low, high);
            while(low <high && arr[low] <= pivotKey){
                low++;
            }
            Swap(arr, low, high);
        }
        return low;
    }
    
    
    private void Swap(int[] arr, int low, int high) {
		int temp = arr[low];
		arr[low] = arr[high];
		arr[high] = temp;
	}
}
```

```
/**思路
*    添加集合，排序，选出前k个数
*/
import java.util.ArrayList;
import java.util.List;
import java.util.Collections;
public class Solution {
    public ArrayList<Integer> GetLeastNumbers_Solution(int [] input, int k) {
        ArrayList<Integer> list=new ArrayList<Integer>();
        if(input==null ||input.length<=0
          ||k<1 ||k>input.length) 
            return list;
        //1 添加
        for(int i=0;i<input.length;i++){
            list.add(input[i]);
        }
        //2 排序
        Collections.sort(list);
        //3 选出前k个数
        while(list.size()>k){
            list.remove(k);
        }
        return list;     
    }
}
```
